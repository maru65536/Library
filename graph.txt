入力
n,m=map(int,input().split())
a=[]
for i in range(n):
    a.append([])
for i in range(m):
    p,q=map(int,input().split())
    a[p-1].append(q-1)
    a[q-1].append(p-1)

深さ優先探索(木の上で！閉路があると無限ループ)
dist=[-1]*n
dist[0]=0 #初期位置のコストを0に。初期位置は変更可能
def dis(a,p):
    for i in range(len(a[p])):
        if dist[a[p][i]]==-1:
            dist[a[p][i]]=dist[p]+1
            dis(a,a[p][i])
dis(a,0) #初期位置を変更した場合はこっちも
print(dist)
